608

613 
614        // For each state, check to see if it is an Orthogonal state. If so,

616        // only then get its Orthogonal Regions.
617        for (Map.Entry<String, State> entry : states.entrySet()) {
618            State state = entry.getValue();

620            // If the state is not an Orthogonal state, skip it
621            if (!state.isOrthogonal())

623

626                continue;

629            for (Region region : state.getRegion()) {
630                Map.Entry<String, Region> ns = getNamespace(entry.getKey(), region);
631                regions.put(ns.getKey(), region);
632            }

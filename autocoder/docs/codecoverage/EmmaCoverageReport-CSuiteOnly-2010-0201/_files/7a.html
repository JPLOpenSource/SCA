420    private static Collection<String> searchRegularDirectory (String packagePath, final Pattern p) {
421        // regular file structure, recursively seek classes to add to list
422        List<String> foundList = new ArrayList<String>();
423        Queue<File> searchQue = new LinkedList<File>();
424        /* because we're search-replacing a File path using packagePath, need
425         * to do special handling for DOS; we might start with packagePath
426         * "/X:/path/to/package" or "//srv/path/to/package", which need to
427         * sanitize to become "[X:]\path\to\package..." */
428        if (! File.separator.equals("/")) {
429                // remove leading '/' in "/X:" and then replace '/' with '\\'
430                packagePath = packagePath.replaceFirst("^/(\\w+:)", "$1").replace("/", File.separator);
431        }  // now we should have packagePath "[X:]\path\to\package"
432        /* make sure package path ends with a separator so the matched filenames
433         * properly get converated to class names. */
434        if (! packagePath.endsWith(File.separator)) {
435                packagePath = packagePath + File.separator;
436        }
437 
438        searchQue.add(new File(packagePath));
439        while (!searchQue.isEmpty()) {
440            File searchDir = searchQue.poll();
441            Set<File> containedFiles = new HashSet<File>();  // all children
442            Collections.addAll(containedFiles, searchDir.listFiles());
443            Set<File> matchedFiles = new HashSet<File>();  // matching children
444            Collections.addAll(matchedFiles, searchDir.listFiles(new FilenameFilter() {
445                public boolean accept (File dir, String name) {
446                    // find all matching classes
447                    return p.matcher(name).matches();
448                }
449            }));
450            for (File f : matchedFiles) {
451                // remove relative path to package, then replace seps with '.'
452                foundList.add(f.getPath().replace(packagePath, "").replace(File.separator, "."));
453            }
454            // find all subdirectories to search under
455            containedFiles.removeAll(matchedFiles);  // avoid duplicate search
456            for (File f : containedFiles) {
457                if (f.isDirectory()) {
458                    searchQue.offer(f);
459                }
460            }
461        }
462        return foundList;
463    }

465    /**
466     * Searches and returns within the classpaths the list of resource paths
467     * containing the supplied Class, inclusive of package directories.
468     * <p>
469     * In the current implementation, if application is executed within a
470     * directory context (i.e., development environment), it'll simply return
471     * the directory containing &lt;package&gr;<code>clazz</code>.
472     * Otherwise, when executed in a Jar context (i.e., deployment), it'll
473     * return the Jar file containing <code>clazz</code>, as matched by the pattern
474     * <blockquote>
475     * <code>^(.*)&lt;packageAsDir&gt;/&lt;rname$&gt;</code>
476     * </blockquote>where '/' is actually {@link File#separator}.
477     * </p>
478     * @param clazz  the class whose resource to find
479     * @param rname  name of resource sought
480     * @return  String array of resource paths
481     */
482    public static String[] resourcePaths (Class<?> clazz, String rname) {
483        List<String> pathList = new ArrayList<String>();
484        String packageName = clazz.getPackage().getName();
485        // construct a regex pattern that captures everything before package
486        String fullPattern = "^(.*)" + packageName.replace(".", "/")
487                + "/" + rname + ".*";
488        final Pattern p = Pattern.compile(fullPattern);
489 
490        // collect the set of paths to search from classpath
491        List<String> searchList = new ArrayList<String>();
492        searchList.add(".");  // add current path in case it's a regular dir
493        String userdir = System.getProperty("user.dir");
494        StringTokenizer tokens = new StringTokenizer(System.getProperty("java.class.path"), File.pathSeparator);
495        while (tokens.hasMoreTokens()) {
496            String path = tokens.nextToken();
497            // replace all DOS file-seps with Unix file-sep 
498            searchList.add(path.replaceAll("\\\\", "/"));
499        }
500 
501        // now search all the candidate paths
502        for (String path : searchList) {
503            if (path.equals(".")) {
504                URL packageURL = clazz.getResource(rname);
505                if (packageURL != null) {
506                    Matcher m = p.matcher(packageURL.getPath());
507                    if (m.matches()) {
508                        pathList.add(m.group(1));
509                    }
510                }
511            } else {  // try looking inside Jar!
512                String abspath = null;  // make path absolute first
513                if (new File(path).isAbsolute()) {  // already an absolute path
514                    abspath = path;
515                } else {  // make relative to user path
516                    try {
517                        abspath = new File(userdir + File.separator + path).getCanonicalPath();
518                    } catch (IOException e) {
519                        abspath = path;  // can't do much more
520                    }
521                }
522                if (path.endsWith(".jar")) {  // look inside this jar for this class
523                    String jarpath = String.format("jar:file:%s!/", abspath);
524                    try {
525                        JarURLConnection conn = (JarURLConnection)new URL(jarpath).openConnection();
526                        JarFile jarfile = conn.getJarFile();
527                        for (JarEntry je : Collections.list(jarfile.entries())) {
528                            if (p.matcher(je.getName()).matches()) {
529                                pathList.add(jarpath);  // done
530                                break;
531                            }
532                        }
533                    } catch (MalformedURLException e) {  // ignore error
534                    } catch (IOException e) {  // ignore error
535                    }
536                } else {
537                    Matcher m = p.matcher(abspath);
538                    if (m.matches()) {
539                        pathList.add(m.group(1));
540                    }
541                }
542            }
543        }
544 
545        pathList.add(userdir);  // finally, look inside working dir for overrides
546        return pathList.toArray(new String[0]);
547    }

381                                tMapper.incIndent();
382                                // take the else junction transition!
383                                makeStateTransition(w, tgtState, juncTrans);
384                                tMapper.decIndent();
385                            } else {  // uh-oh, unknown!
386                                writeCodePart(w, QPPY_TRANS_DECISION_UNKNOWN);
387                            }
388                        }
389                    }
390                    guardFlagStack.pop();  // restore last guard state in stack
391                    break;
392                case exitPoint:
393                    writeCodePart(w, QPPY_TRANS_EXITPT);
394                    break;
395                case terminate:
396                    writeCodePart(w, QPPY_TRANS_TERMINATE);
397                    break;
398                default:
399                    writeCodePart(w, QPPY_TRANS_UNKNOWN);
400                    break;
401                }
402            } else if (tgtState instanceof ConnectionPointReference) {
403                writeCodePart(w, QPPY_TRANS_CONNPTREF);
404            } else if (tgtState instanceof State
405                    || tgtState instanceof FinalState) {
406                writeCodePart(w, QPPY_TRANS_STATE);
407            } else {  // unsupported/unknown target state type
408                writeCodePart(w, QPPY_TRANS_UNKNOWN);
409            }
410            tContext.remove("target");
411        }
412 
413        if (tModel.hasGuard(transition)) {
414            writeCodePart(w, QPPY_TRANS_GUARD_END);
415            tMapper.decIndent();
416        }
417        tContext.remove("if");
418    }
419 
420    private void makeAction (Writer w, Behavior activity) {
421        if (activity != null) {
422            for (String actionStr : activity.actionList()) {
423                if (Util.isFunctionCall(actionStr)) {
424                    // treat action as function call
425                    tContext.put("funcCall", tModel.getCallParts(actionStr));
426                    writeCodePart(w, QPPY_ACTION_FUNCTION);
427                    tContext.remove("funcCall");
428                } else {  // Event action, so publish as signal event
429                    tContext.put("action", actionStr);
430                    writeCodePart(w, QPPY_ACTION_EVENT);
431                    tContext.remove("action");
432                }
433            }
434        }
435    }

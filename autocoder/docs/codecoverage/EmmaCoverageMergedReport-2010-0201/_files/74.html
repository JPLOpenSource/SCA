181    /**
182     * Computes the sequence of exit/entry Activity along the path of transition
183     * from the <code>source</code> State to the <code>target</code> State.
184     * The algorithm is simply as follows:<ol>
185     * <li> Find the least-common-ancestor (LCA)
186     * <li> Collect the exit actions from the source outward to the LCA
187     * <li> Collect the entry actions from the LCA into the target state
188     * </ol>
189     * @param source  source UML Vertex of transition
190     * @param target  target UML Vertex of transition
191     * @return  List of exit/entry Behaviors of States
192     */
193    public List<Behavior> computeTransitionPathActivity (Vertex source, Vertex target) {
194        List<Behavior> actions = Util.newList();
195 
196        /* Find the LCA:
197         * a.) Determine the depth of source (sD) and target (tD), and pick the
198         *     deepest index 'common' less than or equal to both.
199         * b.) Starting from 'common', compare that ancestor in the ancestor
200         *     chain of source and target, respectively, until one is found that
201         *     is equivalent; that is the LCA.
202         */
203        List<State> sChain = getAncestorChain(source);
204        int sD = sChain.size();
205        List<State> tChain = getAncestorChain(target);
206        int tD = tChain.size();
207        int common = Math.min(sD, tD);  // deepest common depth
208        int lca = -1;  // -1 represents tree root of State Machine
209        for (int d=common-1; d >= 0; --d) {  // array index is one off from depth
210            if (sChain.get(d).equals(tChain.get(d))) {  // found lca!
211                lca = d;
212                break;
213            }
214        }
215 
216        if (lca > -1 && sD-1 == lca) {  // need to include exit of source state
217            actions.add(sChain.get(lca).getExit());
218        }
219        // from source, go up its ancestor chain to (but excl.) lca and collect exits
220        for (int d=sD-1; d > lca; --d) {
221            actions.add(sChain.get(d).getExit());
222        }
223 
224        // from lca, go down target's ancestor chain to (but excl.) target and collect entries
225        for (int d=lca+1; d < tD-1; ++d) {
226            actions.add(tChain.get(d).getEntry());
227        }
228 
229        return actions; 
230    }

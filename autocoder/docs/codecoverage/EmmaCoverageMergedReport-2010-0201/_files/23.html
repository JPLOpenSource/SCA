54/**
55 * This class supports the validation of a UML Model.  Only critical validation
56 * errors cause autocoding to terminate; all other validation problems report
57 * a warning message to the user.
58 * <p>
59 * The approach of validation is to traverse the Model tree depth-first and
60 * validate each vertex.  A validator method is easily declared by adding it as
61 * a public member declaring the @{@link Validate} annotation (optionally
62 * indicating an order), and specifying one parameter of the type of vertex it
63 * is validating.  Unordered {@link Validate} methods are done after the ordered
64 * ones, in whatever order was returned by the Java reflect infrastructure.
65 * Methods not specifying a parameter are invoked for every types of Vertex.
66 * Methods specifying any other kinds of parameters will be ignored (with warning).
67 * </p>
68 * Algorithm:<ol>
69 * <li> When this class is first instantiated, it collects all the validator
70 * methods and hashes them into a hash of list by the type of Vertex they
71 * validate, ordered in each list by {@link Validate#order()}.
72 * <li> Upon visiting a vertex in the tree, all {@link Validate} methods
73 * applicable to that vertex are invoked in order, error message are reported
74 * in the process, and any return value reported.
75 * <li> Fatal errors do not terminate the validation process immediately, but
76 * are flagged to cause the {@link #validate(Model)} method to throw a
77 * {@link FatalModelException} at the very end of the validation process.
78 * This allows all problems in the model to be reported in one go.
79 * </ol>
80 * TODO Errors to check on: <ul>
81 * <li> Do we need to ensure that destination state of initial transition is a child of the composite parent state?

492    /**
493     * Looking only at a state's outgoing transitions, checks to make sure
494     * that a transition is OK: <ul>
495     * <li> makes sure no duplicate signal events is found on two out-transitions
496     * <li> makes sure transition target state is resolvable
497     * <li> makes sure all signal events resolve to a signal object
498     * <li> makes sure all transition signals resolve to a signal object
499     * <li> makes sure all time events have a timer expression
500     * </ul>
501     * N.B.: This means events that aren't used are ignored.
502     * 
503     * @param state  state whose outgoing transitions to check
504     */
505    @Validate
506    public void checkTransitionOK (State state) {
507        Set<String> signalSeen = Util.newSet();
508        for (Transition transition : state.getOutgoing()) {
509            // check transition target resolves to a state
510            checkTransitionTarget(state, transition);
511 
512            // check triggers
513            boolean hasTimeout = false;
514            boolean hasSignal = false;
515            String signalName = null;
516            for (Trigger t : transition.getTrigger()) {
517                if (t.getEvent() instanceof SignalEvent) {
518                    hasSignal = true;
519                    SignalEvent ev = ((SignalEvent) t.getEvent());
520                    if (checkSignalEvent(state, transition, ev)) {
521                        signalName = ev.getSignal().getName();
522                        if (signalSeen.contains(signalName)) {
523                            // fatal! duplicate out-transition events
524                            setFatalError();
525                            StringBuilder sb = new StringBuilder("Fatal: Duplicate signal event found!");
526                            buildStateTransitionInfoStr(sb, state, transition, ev);
527                            Util.error(sb.toString());
528                        } else {  // mark this signal as seen
529                            signalSeen.add(signalName);
530                        }
531                    }
532                } else if (t.getEvent() instanceof TimeEvent) {
533                    hasTimeout = true;
534                    checkTimeEvent(state, transition, (TimeEvent) t.getEvent());
535                }
536                if (hasSignal && hasTimeout) {  // error!
537                    setFatalError();
538                    Util.error("Fatal: Invalid transition "
539                            + t.id() + " (signal " + signalName
540                            + ") can't have both trigger event and timeout.");
541                    break;  // done with triggers, next transition!
542                }
543            }
544        }
545    }

548        boolean rv = true;
549 
550        Vertex target = t.getTarget();

567            }

569        
570        return rv;
571    }

574        boolean rv = true;
575 
576        Signal signal = ev.getSignal();
577        if (signal == null) {
578            rv = false;
579            setFatalError();

581                // distinguish between NO signal defined at all, or unresolved Signal ID

583                String signalID = Util.getNodeAttribute(((UMLElement) ev).getNode(),

587                } else {  // defined by can't dereference XMI ID!

591                buildStateTransitionInfoStr(msgSB, state, t, ev);
592                Util.error(msgSB.toString());
593            }
594        }

634    private void buildStateTransitionInfoStr (StringBuilder msgSB, State state, Transition t, Event ev) {
635        // info on source state
636        msgSB.append(" (s,t,s') := (").append(state.getName()).append(", ");
637        if (t.isInternal()) {
638            msgSB.append("(internal) ");
639        }
640        // get what label we can for transition
641        if (ev != null && ev.getName() != null) {
642            msgSB.append(ev.getName());
643        } else {
644            Trigger trigger = t.getTrigger().iterator().next();
645            if (trigger != null) {
646                msgSB.append(trigger.getName());
647            } else if (t.getName() != null && !t.getName().equals("")) {
648                msgSB.append(t.getName());
649            } else {
650                msgSB.append("");
651            }
652        }
653        if (t.getEffect() == null) {
654            msgSB.append("");
655        } else {
656            msgSB.append(" / ").append(t.getEffect().getName());
657        }
658        // do target state if available
659        msgSB.append(", ");
660        if (t.getTarget() == null) {
661            msgSB.append("");
662        } else {
663            msgSB.append(t.getTarget().getName());
664        }
665        msgSB.append(")");
666 
667        if (ev != null) {  // also include event ID
668            msgSB.append(", Event ID '").append(ev.id()).append("'");
669        } else {  // include transition ID otherwise
670            msgSB.append(", Transition ID '").append(t.id()).append("'");
671        }
672    }
